    Usage

    Markup

    <div class="switchers" data-mode="accordion">
  <div class="switcher-head">
    <button class="switcher-toggle-all">switcher-toggle-all</button>
  </div>
  <div class="switcher">
    <div class="switcher-header">
      <div class="switcher-header-col">
        switcher-header-col
      </div>
      <div class="switcher-header-col switcher-toggle">
        switcher-header-col
      </div>
    </div>
    <div class="switcher-body">
      switcher-body
    </div>
  </div><!-- /.switcher -->
  <div class="switcher">
    ...
  </div><!-- /.switcher -->
</div>

    JavaScript

    $('.switchers').switchers({
  switcherClass: 'switcher',
  toggleClass: 'switcher-toggle',
  allToggleClass: 'switcher-toggle-all',
  activeClass: 'switcher-active'
});

    Methods

    $().switchers(options)

    $().switchers('destroy')

    $('#element').switchers('destroy')
